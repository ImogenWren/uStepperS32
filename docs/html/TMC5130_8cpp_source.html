72void TMC5130::enableStallguard(int8_t threshold, bool stopOnStall, float rpm)
73{
74 // Limit threshold
75 if (threshold > 63)
76 threshold = 63;
77 else if (threshold < -64)
78 threshold = -64;
79
80 rpm = abs(rpm);
81 // Limit rpm
82 if (rpm > 1000)
83 rpm = 1000;
84 else if (rpm < 2)
85 rpm = 2;
86
87 /* Disable StealthChop for stallguard operation */
88 this->writeRegister(GCONF, EN_PWM_MODE(0) | I_SCALE_ANALOG(1));
89 this->setShaftDirection(ptr->shaftDir);
90
91 // Configure COOLCONF for stallguard
92 this->writeRegister(COOLCONF, SGT(threshold) | SFILT(1) | SEMIN(5) | SEMAX(2) | SEDN(1));
93
94 //int32_t stall_speed = 1048576 / ptr->rpmToVelocity * speed // 1048576 = 2^20. See TSTEP in datasheet p.33
95 int32_t stall_speed = 1048576 / ptr->rpmToVelocity * (rpm / 2); //Should be 1048576 = 2^20.
96 stall_speed = stall_speed * 1.2; // // Activate stallGuard sligthly below desired homing velocity (provide 20% tolerance)
97
98 // Set TCOOLTHRS to max speed value (enable stallguard for all speeds)
99 this->writeRegister(TCOOLTHRS, stall_speed); // Max value is 20bit = 0xFFFFF
100 this->writeRegister(THIGH, 0);
101
102 // Enable automatic stop on stall dectection
103 if (stopOnStall)
104 this->writeRegister(SW_MODE, SG_STOP(1));
105 else
106 this->writeRegister(SW_MODE, SG_STOP(0));
107}

261void TMC5130::setRampMode(uint8_t mode)
262{
263
264 switch (mode)
265 {
266 case POSITIONING_MODE:
267 // Positioning mode
268 this->writeRegister(VSTART_REG, this->VSTART);
269 this->writeRegister(A1_REG, this->A1);

277 break;

281 this->writeRegister(VSTART_REG, this->VSTART);

354int32_t TMC5130::writeRegister(uint8_t address, uint32_t datagram)
355{
356 semaphore.getLock();
357 // Enable SPI mode 3 to use TMC5130
358 LL_SPI_SetClockPhase(this->spiHandle._spiChannel, LL_SPI_PHASE_2EDGE);
359 LL_SPI_SetClockPolarity(this->spiHandle._spiChannel, LL_SPI_POLARITY_HIGH);
360
361 uint32_t package = 0;
362
363 // Add the value of WRITE_ACCESS to enable register write
364 address += 0x80;
365
366 this->spiHandle.csReset();
367
368 this->status = this->spiHandle.transmit8BitData(address);
369
370 package |= this->spiHandle.transmit8BitData((datagram >> 24) & 0xff);
371 package <<= 8;
372 package |= this->spiHandle.transmit8BitData((datagram >> 16) & 0xff);
373 package <<= 8;
374 package |= this->spiHandle.transmit8BitData((datagram >> 8) & 0xff);
375 package <<= 8;
376 package |= this->spiHandle.transmit8BitData((datagram)&0xff);
377
378 this->spiHandle.csSet();
379
380 semaphore.releaseLock();
381
